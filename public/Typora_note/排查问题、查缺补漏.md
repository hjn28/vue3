function jsonp(url, callback, error) {
  const scriptTag = document.createElement('script');
  // scriptTag.src = `${url}`;
  const callbacFunc = `jsonpreqcallback_${new Date().getTime()}`;
  scriptTag.src = `${url}&callback=${callbacFunc}&jsoncallback=${callbacFunc}`;
  scriptTag.onerror = () => {
    error && error();
  };
  window[callbacFunc] = function (data) {
    if (callback) {
      callback(data);
    }
  };
  scriptTag.onload = () => {
    if (isMSIE()) {
      scriptTag.removeNode(true);
    } else {
      scriptTag.remove();
    }
  };
  document.querySelector('head').appendChild(scriptTag);
}
